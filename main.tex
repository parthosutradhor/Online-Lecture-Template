	\Large

	\begin{Heading}
		Introduction to Galois Theory
	\end{Heading}

	\begin{Definition}
		[Field Extension] Let $F \subseteq K$ be fields. We say that $K$ is a \emph{field
		extension} of $F$, denoted $K/F$, if the operations of $F$ are the restrictions
		of those of $K$.
	\end{Definition}

	\vspace{5cm}

	\begin{Definition}
		[Automorphism] An \emph{automorphism} of a field $K$ is a bijective map
		$\sigma:K \to K$ such that
		\[
			\sigma(a+b) = \sigma(a)+\sigma(b), \quad \sigma(ab) = \sigma(a)\sigma(b), \quad
			\sigma(1)=1.
		\]
	\end{Definition}

	\clearpage

	\begin{Theorem}
		[Fundamental Theorem of Galois Theory] Let $K/F$ be a finite Galois extension
		with Galois group $G=\mathrm{Gal}(K/F)$. Then there is a one-to-one
		correspondence between:
		\begin{itemize}
			\item Intermediate fields $F \subseteq E \subseteq K$,

			\item Subgroups $H \leq G$,
		\end{itemize}
		given by $E \mapsto \mathrm{Gal}(K/E)$ and $H \mapsto K^{H}$.
	\end{Theorem}

	\vspace{6cm}

	\begin{Problem}
		[Example Task] Show that the extension $\mathbb{Q}(\sqrt{2})/\mathbb{Q}$ is Galois.
		Find its Galois group explicitly.
	\end{Problem}

	\clearpage

	\begin{Heading}
		Worked Example
	\end{Heading}

	\begin{Definition}
		[Splitting Field] A \emph{splitting field} of a polynomial $f(x) \in F[x]$
		is the smallest field extension $K/F$ in which $f(x)$ factors into linear
		terms.
	\end{Definition}

	\vspace{6cm}

	\begin{Theorem}
		[Galois Group of $x^{2}-2$] Let $f(x)=x^{2}-2$ over $\mathbb{Q}$. Then the
		splitting field is $\mathbb{Q}(\sqrt{2})$, and
		\[
			\mathrm{Gal}\big(\mathbb{Q}(\sqrt{2})/\mathbb{Q}\big) \cong C_{2}.
		\]
	\end{Theorem}

	\clearpage

	\begin{Problem}
		Compute the Galois group of $x^{3}-2$ over $\mathbb{Q}$. (Hint: It should be
		isomorphic to a subgroup of $S_{3}$.)
	\end{Problem}
