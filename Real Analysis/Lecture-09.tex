\begin{heading}
	Subsequence
\end{heading}

\vspace{10pt}

\begin{definition}[Subsequence]
	Let $(a_n)$ be a sequence of real numbers, and let 
	\[ n_1 < n_2 < n_3 < n_4 < n_5 < \cdots \]
	be an increasing sequence of natural numbers. Then the sequence
	\[ a_{n_1}, a_{n_2}, a_{n_3}, a_{n_4}, a_{n_5}, \cdots \]
	is called a \textit{subsequence} of $(a_n)$ and is denoted by $(a_{n_j})$, 
	where $j \in \mathbb{N}$ indexes the subsequence.
\end{definition}

\clearpage

\begin{heading}
	Examples of Subsequence
\end{heading}

\clearpage

\begin{theorem}
	Subsequences of a convergent sequence converge to the same limit as the original sequence.
\end{theorem}

\clearpage

\begin{heading}
	Bolzano–Weierstrass Theorem
\end{heading}

\vspace{10pt}

\begin{definition}[Bolzano–Weierstrass Theorem]
	\textbf{Every bounded sequence contains a convergent subsequence.}
\end{definition}

\clearpage

\begin{theorem}[NIP $\implies$ BWT]
	Let $(a_n)$ be a bounded sequence. Using the Nested Interval property, show that there exists a convergent subsequence.
\end{theorem}

\clearpage

\begin{theorem}[Aoc $\implies$ BWT]
	Let $(a_n)$ be a bounded sequence. Using the Axiom of Completeness, show that there exists a convergent subsequence.
\end{theorem}

\clearpage

\begin{theorem}
	Assume $(a_n)$ is a bounded sequence with the property that every convergent subsequence of $(a_n)$ converges to the same limit $a \in \R$. Show that $(a_n)$ must converge to $a$.
\end{theorem}
