    \begin{heading}
		$\mathbb{Q}$ is not complete
	\end{heading}

    \vspace{10pt}

    \begin{definition}[Completeness]
    A field $F$ is said to be \textbf{complete} if every nonempty subset $A \subseteq F$ that is bounded above 
    has a least upper bound (supremum) in $F$.
    \end{definition}

    \vspace{10pt}

    \begin{problem}
        Show that the set $A = \{ x \in \mathbb{Q} : x^2 < 2 \}$ is bounded above in $\mathbb{Q}$ but does not have a supremum in $\mathbb{Q}$.     
    \end{problem}

    \clearpage

    \begin{heading}
		The ways to complete $\mathbb{Q}$
	\end{heading}

    \clearpage

    \begin{heading}
		Dedekind Cuts
	\end{heading}

    \begin{definition}[Dedekind Cut]
        A \textbf{cut} $\Sigma$ is a subset of $\mathbb{Q}$ such that:
        \begin{enumerate}
            \item $\Sigma \neq \varnothing$ and $\Sigma \neq \mathbb{Q}$ \hfill \textit{(non-trivial)}
            \item If $p \in \Sigma$, $q \in \mathbb{Q}$, and $q < p$, then $q \in \Sigma$ \hfill \textit{(closed downward)}
            \item If $p \in \Sigma$, then $ \exists r \in \Sigma$ with $p < r$ \hfill \textit{(no maximum element)}
        \end{enumerate}
    \end{definition}

    \clearpage

    \begin{heading}
		Examples of Cuts
	\end{heading}

    \begin{problem}[Example]
        Show that
        $$\{ x \in \mathbb{Q} : x < 2 \}$$
        is Dedekind cut.
    \end{problem}
    \vfill
    \begin{problem}[Non-example]
        Show that
        $$\{ x \in \mathbb{Q} : x <= 2 \}$$
        is not a Dedekind cut.
    \end{problem}
    \vfill
    \clearpage
    \begin{problem}[Example]
        Show that
        $$\{ x \in \mathbb{Q} : x^2 < 2\}$$
        is not Dedekind cut.
    \end{problem}
    \vfill
    \begin{problem}[Non-example]
        Show that
        $$\{ x \in \mathbb{Q} : x^2 < 2 \text{ or } x < 0 \}$$
        is Dedekind cut.
    \end{problem}\
    \vfill
    \clearpage

    \begin{heading}
		What is the set of real number?
	\end{heading}

    \begin{definition}[Definition of $\mathbb{R}$]
        The set of real numbers $\mathbb{R}$ is defined as the set of all Dedekind cuts in $\mathbb{Q}$. Mathematically,
        $$\mathbb{R} = \{ \Sigma : \Sigma \text{ is a Dedekind cut in } \mathbb{Q} \}.$$
    \end{definition}

    \clearpage

    \begin{heading}
		Addition of two Cuts
	\end{heading}
    \vspace{10pt}
    \begin{definition}[Addition of Cuts]
        If $\Sigma$ and $\Gamma$ are two cuts, then we define their sum as
        $$\Sigma + \Gamma = \{ a + b : a \in \Sigma, b \in \Gamma \}.$$
    \end{definition}
    \begin{theorem}[Well-definedness of Addition]
        If $\Sigma$ and $\Gamma$ are two cuts, then $\Sigma + \Gamma$ is also a cut.
    \end{theorem}

    \clearpage

    \begin{heading}
		Negative of a Cuts
	\end{heading}
    \vspace{10pt}
    \begin{definition}[Negative of a Cuts]
        If $\Sigma$ is a cut, then we define its negative as
        \[-\Sigma = \{ x \in \mathbb{Q} : -x \notin \Sigma \text{ and } -x \text{ is not the least element of } \mathbb{Q} \setminus \Sigma \}.\]
        or
		\[-\Sigma = \{ x \in \mathbb{Q} : \exists r > 0 \text{ s.t. } -p - r \notin \Sigma \}\]
    \end{definition}
    \begin{theorem}[Well-definedness of Negative]
        If $\Sigma$ is a cut, then $-\Sigma$ is also a cut.
    \end{theorem}

    \clearpage

    \begin{definition}[Multiplication of Cuts]
        If $\Sigma$ and $\Gamma$ are two cuts, then we define their product as
        $$\Sigma \cdot \Gamma = \{ a \cdot b : a \in \Sigma, b \in \Gamma, a > 0, b > 0 \}.$$
    \end{definition}
    \begin{theorem}[Well-definedness of Multiplication]
        If $\Sigma$ and $\Gamma$ are two cuts, then $\Sigma \cdot \Gamma$ is also a cut.
    \end{theorem}

    \clearpage

    \begin{heading}
		Field Axioms of $\mathbb{R}$
	\end{heading}
    \vspace{10pt}

    \clearpage

    \begin{heading}
		Ordering on $\mathbb{R}$
	\end{heading}
    \vspace{10pt}
    \begin{definition}[Ordering on Cuts]
        If $\Sigma$ and $\Gamma$ are two cuts, then we define
		\[\Sigma < \Gamma \iff \Sigma \subsetneq \Gamma.\]
        
    \end{definition}

	\clearpage

	\begin{heading}
		Facts about real numbers
	\end{heading}

    \vspace{10pt}

    \begin{theorem}[Least Upper Bound Property]
        Every nonempty subset of $\mathbb{R}$ that is bounded above has a least upper bound in $\mathbb{R}$.
    \end{theorem}

    \vspace{10pt}

	\begin{theorem}[Gratest Lower Bound Property]
        Every nonempty subset of $\mathbb{R}$ that is bounded below has a greatest lower bound in $\mathbb{R}$.
    \end{theorem}

    \vspace{10pt}

    \begin{theorem}[Uniqueness of $\mathbb{R}$]
        $\mathbb{R}$ is the only complete ordered field (up to isomorphism) containing $\mathbb{Q}$.
    \end{theorem}
