    \begin{heading}
		The Absolute Value Function
	\end{heading}

    \vspace{10pt}

    \begin{definition}[Absolute Value]
		For any real number $x$, the \textbf{absolute value} of $x$, denoted by $|x|$, is defined as
		\[
			|x| = \begin{cases}
				x, & \text{if } x \geq 0, \\
				-x, & \text{if } x < 0.
			\end{cases}
		\]
    \end{definition}

	\vspace{10pt}

	\begin{theorem}
		Show that for any real numbers $x$,  the following holds:
		$$\sqrt{x^2}=|x|$$
	\end{theorem}

    \clearpage

	\begin{heading}
		The Triangle Inequality
	\end{heading}

    \vspace{10pt}

	\begin{theorem}[The Triangle Inequality for Real Numbers]
		For any real numbers $x$ and $y$, the following inequality holds:
		$$|x+y| \leq |x| + |y|$$
	\end{theorem}

    \clearpage

	\begin{theorem}[Other forms of the Triangle Inequality]
		For any real numbers $x$ and $y$, the following inequalities hold:
		$$|x-y| \leq |x| + |y|$$
		$$||x| - |y|| \leq |x-y|$$
	\end{theorem}

    \clearpage

	\begin{heading}
		Equality of two Real Numbers
	\end{heading}

    \vspace{10pt}

	\begin{theorem}
		Two real numbers $a$ and $b$ are equal if and only if for every $\varepsilon > 0$,
		$$|a - b| < \varepsilon$$
	\end{theorem}

    \clearpage

	\begin{heading}
		Axiom of Completeness (AoC)
	\end{heading}

    \vspace{10pt}

	\begin{definition}[Least Upper Bound (Supremum)]
	A real number $s$ is the \textit{least upper bound} for a set $A \subseteq \mathbb{R}$ if it meets the following two criteria:
	\begin{enumerate}[label=(\roman*)]
		\item $s$ is an upper bound for $A$;
		\item if $u$ is any upper bound for $A$, then $s \leq u$.
	\end{enumerate}
	\end{definition}

	\vspace{10pt}

	\begin{definition}[Axiom of Completeness (AoC)]
		Every non-empty set of real numbers that is bounded above has a least upper bound (supremum) in $\mathbb{R}$.
	\end{definition}

	\clearpage

	\begin{definition}[Greatest Lower Bound (Infimum)]
	A real number $t$ is the \textit{greatest lower bound} for a set $A \subseteq \mathbb{R}$ if it meets the following two criteria:
	\begin{enumerate}[label=(\roman*)]
		\item $t$ is a lower bound for $A$;
		\item if $l$ is any lower bound for $A$, then $t \geq l$.
	\end{enumerate}
	\end{definition}

	\vspace{10pt}

	\begin{theorem}[Theorem: Existence of Infimum]
		Show that every non-empty set of real numbers that is bounded below has a greatest lower bound (infimum) in $\mathbb{R}$.
	\end{theorem}

    \clearpage

	\begin{theorem}[Theorem: Characterization of Supremum]
		Assume $s \in \mathbb{R}$ is an upper bound for a set $A \subseteq \mathbb{R}$. 
		Then, $s = \sup A$ if and only if, for every choice of $\epsilon > 0$, there exists an element $a \in A$ satisfying 
		$$s - \epsilon < a$$
	\end{theorem}

    \clearpage

	\begin{theorem}[Theorem: Characterization of Infimum]
		Assume $t \in \mathbb{R}$ is a lower bound for a set $A \subseteq \mathbb{R}$. 
		Then, $t = \inf A$ if and only if, for every choice of $\epsilon > 0$, there exists an element $a \in A$ satisfying 
		$$a < t + \epsilon$$
	\end{theorem}

	\clearpage

	\begin{heading}
		Consequences of Completeness (AoC)
	\end{heading}

	\clearpage

	\begin{heading}
		Nested Interval Property (NIP)
	\end{heading}

	\vspace{10pt}

	\begin{definition}[Nested Interval Property (NIP)]
	For each $n \in \mathbb{N}$, assume we are given a closed interval 
	$I_n = [a_n, b_n] = \{\, x \in \mathbb{R} : a_n \le x \le b_n \,\}$. 
	Assume also that each $I_n$ contains $I_{n+1}$. Then, the resulting 
	nested sequence of closed intervals
	\[
	I_1 \supseteq I_2 \supseteq I_3 \supseteq I_4 \supseteq \cdots
	\]
	has a nonempty intersection; that is,
	\[
	\bigcap_{n=1}^{\infty} I_n \neq \varnothing.
	\]
	\end{definition}

	\vspace{10pt}

	\begin{theorem}[Theorem: AoC $\implies$ NIP]
		The Axiom of Completeness implies the Nested Interval Property.
	\end{theorem}

	\clearpage

	\begin{heading}
		Archimedean Property (AP)
	\end{heading}

	\vspace{10pt}
	
	\begin{definition}[Archimedean Property (AP)]
	Given any number $x \in \mathbb{R}$ and any real number $y > 0$, the following two statements hold:
		\begin{itemize}
		\item[(i)] \textit{There exists an $n \in \mathbb{N}$ satisfying $n > x$.}
		\item[(ii)] \textit{There exists an $n \in \mathbb{N}$ satisfying $1/n < y$.}
	\end{itemize}
	\end{definition}

	\vspace{10pt}

	\begin{theorem}[Theorem: AoC $\implies$ AP]
		The Axiom of Completeness implies the Archimedean Property.
	\end{theorem}

	\clearpage

	\begin{heading}
		NIP + AP $\implies$ AoC
	\end{heading}

	\vspace{10pt}

	\begin{theorem}[Theorem: NIP + AP $\implies$ AoC]
		The Nested Interval Property and the Archimedean Property together imply the Axiom of Completeness.		
	\end{theorem}

	\clearpage

	\begin{heading}
		Density of Rationals in Reals
	\end{heading}

	\vspace{10pt}
	
	\begin{theorem}[Density of $\mathbb{Q}$ in $\mathbb{R}$]
		\textit{For every two real numbers $a$ and $b$ with $a < b$, there exists a rational number $r$ satisfying $a < r < b$.}
	\end{theorem}
