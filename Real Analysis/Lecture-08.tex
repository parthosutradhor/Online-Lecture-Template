\begin{heading}
	Monotone Sequences
\end{heading}

\vspace{10pt}

\begin{definition}[Monotonically Increasing Sequence]
	A sequence $\{a_n\}$ is said to be \textbf{monotonically increasing} if for all natural numbers $n,$ the terms of the sequence satisfy the inequality
	\[
	a_n \leq a_{n+1}.
	\]
\end{definition}

\vspace{10pt}

\begin{definition}[Monotonically Decreasing Sequence]
	A sequence $\{a_n\}$ is said to be \textbf{monotonically decreasing} if for all natural numbers $n,$ the terms of the sequence satisfy the inequality
	\[
	a_n \geq a_{n+1}.
	\]
\end{definition}

\begin{heading}
	Monotone Convergence Theorem (MCT)
\end{heading}

\vspace{10pt}

\begin{theorem}[Monotone Convergence Theorem]
	\textbf{If a sequence is monotone and bounded, then it converges.}
\end{theorem}

\clearpage

\begin{theorem}[Aoc $\implies$ MCT]
	Let $(a_n)$ be a monotone and bounded sequence. Using the Axiom of Completeness, show that the sequence $(a_n)$ converges.
\end{theorem}

\end{heading}

\vspace{10pt}

\begin{definition}
	The \textbf{limit superior} of a bounded sequence $(a_n)$ is defined as
    \[
        \limsup_{n \to \infty} a_n = \lim_{n \to \infty} \left( \sup_{k \ge n} a_k \right).
    \]
\end{definition}

\begin{theorem}
	For every bounded sequence $a_n$,
	\[ \liminf_{n \to \infty} a_n \le \limsup_{n \to \infty} a_n \]
\end{theorem}

\clearpage

\begin{theorem}
	Any sequence $a_n$ converges if and only if
	\[ \liminf_{n \to \infty} a_n = \limsup_{n \to \infty} a_n \]
\end{theorem}
