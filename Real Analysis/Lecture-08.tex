\begin{heading}
	Monotone Sequences
\end{heading}

\vspace{10pt}

\begin{definition}[Monotonically Increasing Sequence]
	A sequence $\{a_n\}$ is said to be \textbf{monotonically increasing} if for all natural numbers $n,$ the terms of the sequence satisfy the inequality
	\[
	a_n \leq a_{n+1}.
	\]
\end{definition}

\vspace{10pt}

\begin{definition}[Monotonically Decreasing Sequence]
	A sequence $\{a_n\}$ is said to be \textbf{monotonically decreasing} if for all natural numbers $n,$ the terms of the sequence satisfy the inequality
	\[
	a_n \geq a_{n+1}.
	\]
\end{definition}

\clearpage

\begin{heading}
	Monotone Convergence Theorem (MCT)
\end{heading}

\vspace{10pt}

\begin{theorem}[Monotone Convergence Theorem]
	\textbf{Every bounded monotone sequence converges.}\\[10pt]
	Specifically,
	\begin{itemize}
		\item If monotonically increasing and bounded above $\implies$ converges to its supremum.
		\item If monotonically decreasing and bounded below $\implies$ converges to its infimum.
		\item If monotone but not bounded $\implies$ diverges to either $\infty$ or $-\infty$.
	\end{itemize}
\end{theorem}
